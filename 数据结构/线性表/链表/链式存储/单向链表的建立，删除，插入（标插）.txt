#include <stdio.h>
#include <stdlib.h>
struct link
{
    int data;
    struct link* next;
};
struct link* Append(struct link* head)
{
    int data;
    struct link* pr = head,*p = NULL;
    printf("Please input the data you want?");
    scanf("%d",&data);
    getchar();
    p = (struct link*)malloc(sizeof(struct link));
    if(p == NULL)
    {
        printf("Not enough memory!");
        exit(0);
    }
    else
    {
        if(head == NULL)
        {
            head = p;
        }
        else
        {
            while(pr->next!=NULL)
            {
                pr = pr->next;
            }
            pr->next = p;
        }
    }
    p->data = data;
    p->next = NULL;
    return head;
}

struct link* Delete(struct link* head)
{
    struct link* pr = head,*p = NULL;
    int emit;
    printf("What data do you want to delete?");
    scanf("%d",&emit);
    getchar();
    if(head == NULL)
    {
        printf("The link is empty!\n");
        return head;
    }
    else
    {
        while(pr->data!=emit&&pr->next!=NULL)
        {
            p = pr;
            pr = pr->next;
        }
        if(pr->data==emit)
        {
            if(pr == head)
            {
                head = pr->next;
                free(pr);
            }
            else
            {
                p->next = pr->next;
                free(pr);
            }
        }
        else
        {
            printf("No found!\n");
        }
    }
    return head;
}
struct link* Insert(struct link* head)
{
    struct link*p = NULL,*pr = head,*temp = NULL;
    int insert;
    printf("What data do you want to insert?");
    scanf("%d",&insert);
    getchar();
    p = (struct link*)malloc(sizeof(struct link));
    if(p == NULL)
    {
        printf("Not enough memory!\n");
        exit(0);
    }
    p->data = insert;
    p->next = NULL;
    if(head == NULL)
    {
        head = p;
        return head;
    }
    else
    {
        while(insert> pr->data&&pr->next != NULL)
        {
            temp = pr;
            pr = pr->next;
        }
        if(pr->data >= insert)
        {
            if(pr == head)
            {
                p->next = head;
            }
            else
            {
                pr = temp;
                p->next = pr->next;
                pr->next = p;
            }
        }
        else
        {
            pr->next = p;
        }
    }
    return head;
}

void Display(struct link* head)
{
    struct link* pr = head;
    int i = 1;
    if(head == NULL)
    {
        printf("The link is empty!\n");
        exit(0);
    }
    else
    {
        while(pr!=NULL)
        {
            printf("%-5d%-5d\n",i,pr->data);
            pr = pr->next;
            i++;
        }
    }
}

void Freelink(struct link* head)
{
    struct link* pr = NULL,*p = head;
    while(p!=NULL)
    {
        pr = p;
        p = p->next;
        free(pr);
    }
}

int main()
{
    struct link* head = NULL;
    char ch;
    printf("Do you want to enter a data to link?(Y/N)");
    scanf("%c",&ch);
    int count = 0;
    while(ch=='Y'||ch=='y')
    {
        getchar();
        count++;
        head = Append(head);
        Display(head);
        printf("Do you want to enter again?(Y/N)");
        scanf("%c",&ch);
    }
    printf("%d data have been appended!\n",count);
    getchar();
    printf("Do you want to delete some data?(Y/N)");
    scanf("%c",&ch);
    count = 0;
    while(ch == 'Y'||ch == 'y')
    {
        getchar();
        count++;
        head = Delete(head);
        Display(head);
        printf("Do you want to delete some data?(Y/N)");
        scanf("%c",&ch);
    }
    getchar();
    printf("%d data have been deleted!\n",count);
    printf("Do you want to insert new data?(Y/N)");
    scanf("%c",&ch);
    count = 0;
    while(ch == 'Y'||ch == 'y')
    {
        count++;
        getchar();
        head = Insert(head);
        Display(head);
        printf("Do you want to insert new data?(Y/N)");
        scanf("%c",&ch);
    }
    printf("%d new data have been inserted!\n",count);
    getchar();
    Freelink(head);
    return 0;
}
