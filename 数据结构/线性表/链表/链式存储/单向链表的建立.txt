#include <stdio.h>
#include <stdlib.h>
struct link
{
    int data;
    struct link* next;
};
struct link* Append(struct link* head)
{
    int data;
    struct link* pr = head,*p = NULL;
    printf("Please input the data you want?");
    scanf("%d",&data);
    getchar();
    p = (struct link*)malloc(sizeof(struct link));
    if(p == NULL)
    {
        printf("Not enough memory!");
        exit(0);
    }
    else
    {
        if(head == NULL)
        {
            head = p;
        }
        else
        {
            while(pr->next!=NULL)
            {
                pr = pr->next;
            }
            pr->next = p;
        }
    }
    p->data = data;
    p->next = NULL;
    return head;
}

void Display(struct link* head)
{
    int i = 1;
    if(head == NULL)
    {
        printf("The link is empty!\n");
        exit(0);
    }
    else
    {
        while(head!=NULL)
        {
            printf("%-5d%-5d\n",i,head->data);
            head = head->next;
            i++;
        }
    }
}

void Freelink(struct link* head)
{
    struct link* pr = NULL,*p = head;
    while(p!=NULL)
    {
        pr = p;
        p = p->next;
        free(pr);
    }
}

int main()
{
    struct link* head = NULL;
    char ch;
    printf("Do you want to enter a data to link?(Y/N)");
    scanf("%c",&ch);
    int count = 0;
    while(ch=='Y'||ch=='y')
    {
        getchar();
        count++;
        head = Append(head);
        Display(head);
        printf("Do you want to enter again?(Y/N)");
        scanf("%c",&ch);
    }
    printf("%d data have been appended!",count);
    Freelink(head);
    return 0;
}
