#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,c,i,j,k;
    scanf("%d%d",&n,&q);
    vector < vector<int> > s(n+1);
    while(q--)
    {
        scanf("%d",&c);
        if(c == 1)
        {
            scanf("%d%d%d",&i,&j,&k);
            if(s[i].size() < j+1)
            {
                s[i].resize(j+1);
            }
            s[i][j] = k;
        }
        else
        {
            scanf("%d%d",&i,&j);
            printf("%d\n",s[i][j]);
        }
    }
    return 0;
}
