#include <bits/stdc++.h>
#include <unordered_map>
using namespace std;
unordered_map<int,bool> s;
int main()
{
    int t,n,a;
    scanf("%d",&t);
    for (int i = 1;i<=t;i++)
    {
        s.clear();
        scanf("%d",&n);
        for (int i = 1;i<=n;i++)
        {
            scanf("%d",&a);
            if(!s[a])
            {
                printf("%d ",a);
                s[a] = true;
            }
        }
        printf("\n");
    }
    return 0;
}
