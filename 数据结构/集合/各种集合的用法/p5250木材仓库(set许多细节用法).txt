{

        if(a == 1)

            if(s.find(b) == s.end())
            {
                s.insert(b);
            }
            else
            {
                printf("Already Exist\n");
            }
        }
        else
        {
            scanf("%d",&b);
            if(s.empty())
            {
                printf("Empty\n");
            }
            else
            {
                if(s.find(b) != s.end())
                {
                    printf("%d\n",b);
                    s.erase(b);
                }
                else
                {
                    it = s.lower_bound(b);
                    x = s.lower_bound(b);
                    if(it == s.begin())
                    {
                        printf("%d\n",*it);
                        s.erase(it);
                    }
                    else
                    {
                        x--;
                        if(abs((*x)-b)<=abs(*(it)-b))
                        {
                            printf("%d\n",*x);
                            s.erase(x);
                        }
                        else
                        {
                            printf("%d\n",*it);
                            s.erase(it);
                        }
                    }
                }
            }
        }
    }
    return 0;
}
