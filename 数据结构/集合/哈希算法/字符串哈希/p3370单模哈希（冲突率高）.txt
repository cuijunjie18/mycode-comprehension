#include <bits/stdc++.h>
using namespace std;
char s[10001];
long long a[10001];
const int mod = 1e8+7;//模数
int base = 347;//进制数
long long Hash(char s[])
{
    long long ans = 0;
    for (int i = 0;s[i];i++)
    {
        ans = (ans*base + s[i])%mod;//单模哈希
    }
    return ans;
}
int main()
{
    int n;
    cin>>n;
    for (int i = 1;i<=n;i++)
    {
        scanf("%s",s);
        a[i] = Hash(s);
    }
    sort(a+1,a+n+1);
    int ans = 1;
    for (int i = 2;i<=n;i++)
    {
        if(a[i]!=a[i-1])
           ans++;
    }
    cout<<ans<<endl;
    return 0;
}
