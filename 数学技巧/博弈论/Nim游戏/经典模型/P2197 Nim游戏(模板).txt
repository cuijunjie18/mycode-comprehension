//背景：代码简单，关键是如何理解证明
#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
        int n,x,ans = 0;
        cin>>n;
        for (int i = 1;i<=n;i++)
        {
            cin>>x;
            ans ^= x;
        }
        if(ans == 0) cout<<"No"<<"\n";
        else cout<<"Yes"<<"\n";
    }
    return 0;
}
