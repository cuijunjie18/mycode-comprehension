#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll n,p;
ll a[3000005],prime[3000005],cnt=0;
ll quick_pow(ll a,ll b)
{
    ll ans = 1;
    while(b)
    {
        if(b&1) ans = ans*a%p;
        a = a*a%p;
        b >>= 1;
    }
    return ans;
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    ll ans=1;
    ll m,s=0;
    cin >> n >> p;
    for(int i = 2; i<=n*2; i++) // 欧拉筛法
    {
        if(a[i] == 0) prime[++cnt] = i;
        for(int j = 1; j <= cnt && prime[j]*i <= n*2; j++)
        {
            a[prime[j]*i] = 1;
            if(i%prime[j] == 0) break;
        }
    }
    for(int i = 1; i <= cnt; i++) // 卡特兰数取模模板
    {
        s = 0;
        m = 2*n;
        while(m > 0)
        {
            m = m/prime[i];
            s = s + m;
        }
        m = n;
        while(m > 0)
        {
            m = m/prime[i];
            s = s - m;
        }
        m = n + 1;
        while(m > 0)
        {
            m = m/prime[i];
            s = s-m;
        }
        ans = ans*quick_pow(prime[i],s)%p;
    }
    cout << ans << "\n";
    return 0;
}
