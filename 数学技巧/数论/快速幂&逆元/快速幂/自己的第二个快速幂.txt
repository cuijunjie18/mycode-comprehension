#include <bits/stdc++.h>
using namespace std;
long long g[64];
int main()
{
    memset(g,1,sizeof(g));
    int a,b;
    scanf("%d%d",&a,&b);
    g[1] = a;
    long long number = 2;
    while(number<=62)
    {
        g[number] = g[number-1]*g[number-1];
        number++;
    }
    long long ans = 1;
    for (int i = 1;i<=31;i++)
    {
        if((b>>(i-1))&1)
        {
            ans *= g[i];
        }
    }
    printf("%lld\n",ans);
    return 0;
}
