#include <bits/stdc++.h>
using namespace std;
long long quickpow(int a,int b)
{
    long long ans = 1;
    while(b)
    {
        if(b&1)
            ans *= a;
        a *= a;
        b = b>>1;
    }
    return ans;
}
int main()
{
    long long a,b;
    scanf("%d%d",&a,&b);
    long long ans = quickpow(a,b);
    printf("%lld\n",ans);
    return 0;
}
