#include <bits/stdc++.h>
using namespace std;
long long quickpow(long long a,int b,int p)
{
    if (a == 0) return 0; // 防止底数为0，0取模错误
    long long ans = 1;
    while(b)
    {
        if(b&1)
            ans = ans*a%p;
        a = a*a%p;//此次为a必须开long long 原因
        b = b>>1;
    }
    return ans;
}
int main()
{
    long long a;
    int b,p;
    scanf("%lld%d%d",&a,&b,&p);
    long long ans = quickpow(a,b,p);
    printf("%lld^%d mod %d=%lld",a,b,p,ans);
    return 0;
}
