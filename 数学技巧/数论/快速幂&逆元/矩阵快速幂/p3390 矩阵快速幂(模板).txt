//背景：校赛网络选拔赛，临时学的
//时间复杂度：O(n^3logk)，其中n为构造矩阵的规模，通常小，k为递推的项数
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int md = 1e9 + 7;
const int N = 105;
int temp[N][N];
void Mul(int a[][N],int b[][N],int n)
{
    memset(temp,0,sizeof(temp)); // 清空临时矩阵

    // 矩阵乘法
    for (int i = 1; i <= n; i++){
        for (int j = 1; j <= n; j++){
            for (int k = 1; k <= n; k++){
                temp[i][j] = (temp[i][j] + a[i][k] * b[k][j])%md;
            }
        }
    }

    // 存储矩阵
    for (int i = 1; i <= n; i++){
        for (int j = 1; j <= n; j++){
            a[i][j] = temp[i][j];
        }
    }
}

void Matrix_qpow(int a[][N],int n,int k)
{
    // 答案矩阵初始化为单位矩阵，等价于ans = 1
    int ans[n+1][N];
    memset(ans,0,sizeof(ans));
    for (int i = 1; i <= n; i++) ans[i][i] = 1;
    while (k)
    {
        if (k & 1) Mul(ans,a,n);
        Mul(a,a,n);
        k >>= 1;
    }
    for (int i = 1; i <= n; i++){
        for (int j = 1; j <= n; j++){
            cout << ans[i][j] << " ";
        }
        cout << "\n";
    }
}
signed main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int n,k;
    cin >> n >> k;
    int a[n+1][N];
    for (int i = 1; i <= n; i++){
        for (int j = 1; j <= n; j++){
            cin >> a[i][j];
        }
    }
    Matrix_qpow(a,n,k);
    return 0;
}
