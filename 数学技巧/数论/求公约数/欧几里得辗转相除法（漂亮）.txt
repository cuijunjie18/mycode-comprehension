#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
    if (a > b) swap(a,b);
    int r = b%a;
    while (r)
    {
        b = a;
        a = r;
        r = b%a;
    }
    return a;
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int x,y,z;
    cin >> x >> y >> z;
    cout << gcd(gcd(x,y),z) << "\n";
    return 0;
}
