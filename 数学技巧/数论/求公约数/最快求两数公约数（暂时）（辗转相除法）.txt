#include <stdio.h>
#include <stdlib.h>
int gcd(int a,int b)
{
    int temp;
    if(a==b)
        return a;
    else if(a>b)
    {
        while(a%b)
        {
            temp = a%b;
            a = b;
            b = temp;
        }
        return b;
    }
    else
    {
        while(b%a)
        {
            temp = b%a;
            b = a;
            a = temp;
        }
        return a;
    }
}
int main()
{
    int a,b;
    scanf("%d %d",&a,&b);
    int maxyue = gcd(a,b);
    printf("%d",maxyue);
}
