#include <stdio.h>
#include <stdlib.h>
#include <math.h>
void Isprime(int n)
{
    if(n<=1)
        printf("%d is not a prime!",n);
    if(n==2)
        printf("%d is a prime!",n);
    if(n>2)
    {
        int flag = 1;
        if(n%2==0)
        {
            printf("%d is not a prime!",n);
            exit(0);
        }
        else
        {
            for (int i = 3;i<=sqrt(n);i += 2)
            {
                if(n%i==0)
                {
                    flag = 0;
                    break;
                }
            }
        }
        if(flag)
            printf("%d is a prime!",n);
        else
            printf("%d is not a prime!",n);
    }
}

int main()
{
    int a;
    scanf("%d",&a);
    Isprime(a);
    return 0;
}
