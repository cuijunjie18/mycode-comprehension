#include <bits/stdc++.h>
using namespace std;
long long inv[30000000];//存储1到30000000模某个特定值的逆元
int main()
{
    int a,b;//求1到a模b的逆元,注意b要大于a，一般b为大于a的素数
    scanf("%d%d",&a,&b);
    inv[0] = 0;
    inv[1] = 1;
    for (int i = 2;i<=a;i++)
    {
        inv[i] = b - (b/i)*inv[b%i]%b;
    }
    for (int i = 1;i<=a;i++)
    {
        printf("%lld\n",inv[i]);
    }
    return 0;
}
