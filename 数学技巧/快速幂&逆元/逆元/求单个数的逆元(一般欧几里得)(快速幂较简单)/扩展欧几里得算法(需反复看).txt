#include <bits/stdc++.h>
using namespace std;
int GCD(int a,int b,int &x,int &y)
{
    if(b == 0)
    {
        x = 1;
        y = 0;
        return a;
    }
    int d = GCD(b,a%b,x,y);//第一次到这里才本质开始，b的意义为可能的最大公约数
    int temp = x;
    x = y;
    y = temp - a/b*y;
    return d;
}
int main()
{
    int a,b;
    scanf("%d%d",&a,&b);
    int x,y;
    int yue = GCD(a,b,x,y);
    printf("%d\n%d %d\n",yue,x,y);
    return 0;
}
