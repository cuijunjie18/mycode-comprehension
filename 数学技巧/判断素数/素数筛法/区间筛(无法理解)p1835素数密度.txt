#include <bits/stdc++.h>
using namespace std;
long long prime[5001];
bool number[50001];
bool vis[1000005];
int cnt;
void prework()
{
    number[1] = 1;
    for (int i = 2;i<=50000;i++)
    {
        if(!number[i]) prime[++cnt] = i;
        for (int j = 1;j<=cnt&&i*prime[j]<=50000;j++)
        {
            number[i*prime[j]] = 1;
            if(i%prime[j] == 0) break;
        }
    }
}
int main()
{
    prework();
    int l,r;
    cin>>l>>r;
    int ans = 0;
    for (int i = 1;i<=cnt;i++)
    {
        long long p = prime[i];
        long long start = max((long long)(2),(l + p - 1)/p)*p;
        for (long long j = start;j<=r;j += prime[i]) vis[j-l+1] = true;
    }
    for (int i = 1;i<=r-l+1;i++)
    {
        if(vis[i]) continue;
        ans++;
    }
    if(l == 1) ans--;
    cout<<ans<<endl;
    return 0;
}
