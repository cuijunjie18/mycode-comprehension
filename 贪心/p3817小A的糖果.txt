//背景:洛谷一遍,Atcoder一遍,觉得要存下(洛谷第一篇题解好思路!!!)
//原理:把原来本身就大于x的数变为x,然后尽量让吃掉的糖果可以影响两边,即贪心
#include <bits/stdc++.h>
using namespace std;
int a[100001];//记录每个数
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);//加快输入输出
    int n,x;
    cin>>n>>x;//输入盒子数及阈值x
    long long ans = 0;//初始化答案
    for (int i = 1;i<=n;i++)
    {
        cin>>a[i];
        if(a[i] > x)//如果单独就大于阈值了
        {
            ans += (a[i] - x);//累加答案
            a[i] = x;//变为x
        }
    }
    int p = 2;//初始化位置指针
    while(p<=n)//条件
    {
        if(a[p] + a[p-1] > x)//如果加上左边的要大
        {
            ans += (a[p] + a[p-1] - x);//累加答案
            a[p] -=  (a[p] + a[p-1] - x);//变化自己
            p++;//移动指针
        }
        else//否则
        {
            p++;//继续移动指针
        }
    }
    cout<<ans<<"\n";//输出答案
    return 0;
}
