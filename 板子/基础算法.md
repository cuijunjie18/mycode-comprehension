lowerbound(起点地址,终点地址,参考值)为查找第一个大于等于某数的值

//注意:我的答案数组下标开始为1,且所有操作区间均为闭区间
#include <bits/stdc++.h>
using namespace std;
int temp[200005];
void my_merge(int a[],int s1,int b[],int s2,int temp[])
{
    int k = 0,i = 0,j = 0;/
    while(i<s1&&j<s2)
    {
        if(a[i] <= b[j]) temp[k] = a[i++];
        else temp[k] = b[j++];
        k++;
    }
    for (;i<s1;i++,k++) temp[k] = a[i];
    for (;j<s2;j++,k++) temp[k] = b[j];
}
void merge_sort(int a[],int l,int r)
{
    if(r - l <= 1)
    {
        if(l >= r) return;
        if(a[r] < a[l]) swap(a[r],a[l]);
        return;
    }
    int mid = (l+r)>>1;//分割区间
    merge_sort(a,l,mid-1);
    merge_sort(a,mid,r);
    my_merge(&a[l],mid-l,&a[mid],r-mid+1,temp); // 合并区间
    for (int i = l;i<=r;i++) a[i] = temp[i-l];
}
int main()
{
    int n;
    cin>>n;
    int a[n+1];
    for (int i = 1;i<=n;i++) cin>>a[i];
    merge_sort(a,1,n);
    for (int i = 1;i<=n;i++) cout<<a[i]<<"\n";
    return 0;
}

#### (2)快速排序

```cpp
#include <stdio.h>
#include <stdlib.h>
void Swap(int *a,int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}
void Sortdata(int a[],int begin,int end)
{
    if(begin>end) return;
    int point = a[begin];
    int i = begin;
    int j = end;
    while(i!=j)
    {
        while(a[j] >= point && j > i) j--;
        while(a[i] <= point && i < j) i++;
        if(j > i) Swap(&a[i],&a[j]);
    }
    a[begin] = a[i];
    a[i] = point;
    Sortdata(a,begin,i-1);
    Sortdata(a,i+1,end);
}

void Print(int a[],int n)
{
    for (int i = 0;i<=n-1;i++)
    {
        printf("%d\n",a[i]);
    }
    return;
}

int main()
{
    int n;
    scanf("%d",&n);
    int number[n];
    for (int i = 0;i<=n-1;i++)
    {
        scanf("%d",&number[i]);
    }
    Sortdata(number,0,n-1);
    Print(number,n);
    return 0;
}
```
