#include <stdio.h>
#include <stdlib.h>
void Swap(int* a, int* b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}
int max(int tree[],int n)
{
    int max = tree[0];
    for (int i = 0;i<=n-1;i++)
    {
        if(tree[i]>max)
        {
            max = tree[i];
        }
    }
    return max;
}

int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    int tree[n];
    for (int i = 0;i<=n-1;i++)
    {
        scanf("%d",&tree[i]);
    }
    long long sum = 0;
    int l = 0 ,r = max(tree,n);
    while(l<=r)
    {
        int mid = l + (r-l)/2;
        for (int i = 0;i<=n-1;i++)
        {
            if(tree[i]>mid)
            {
                sum = sum + tree[i] - mid;
            }
        }
        if(sum>=m)
        {
            l = mid + 1;
        }
        else if(sum<m)
        {
            r = mid - 1;
        }
        sum = 0;
    }
    printf("%d",l-1);
    return 0;
}
