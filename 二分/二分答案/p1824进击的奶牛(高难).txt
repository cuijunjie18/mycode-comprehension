#include <bits/stdc++.h>
using namespace std;
int n,c;
int position[100001];
int ans;
bool check(int a)
{
    int number = 0;
    int l = position[1];
    for (int i = 2;i<=n;i++)
    {
        if(position[i]-l<a) number++;
        else l = position[i];
        if(number+c>n) return false;
    }
    return true;
}
int main()
{
    cin>>n>>c;
    for (int i = 1;i<=n;i++)
    {
        cin>>position[i];
    }
    sort(position+1,position+n+1);
    int l = 0;
    int r = position[n]- position[1];
    while(l<=r)
    {
        int mid = (l+r)/2;
        if(check(mid))
        {
            ans = mid;
            l = mid + 1;
        }
        else
        {
            r = mid - 1;
        }
    }
    cout<<ans<<endl;
    return 0;
}
