//背景:第二次刷这题,依旧自己做不对
//原理:二分 + 前缀和
//时间复杂度:o(nlogn)
#include <bits/stdc++.h>
using namespace std;
int n,m;
long long number[100001];//原始数组*1e5
long long sum[100001];//前缀和数组
long long ans;//答案
int r;//最大值
int main()
{
    cin>>n>>m;//输入数组大小及至少区间长度
    for (int i = 1;i<=n;i++)//输入n个数
    {
        cin>>number[i];
        number[i] *= 100000;//扩大
        if(number[i]>r)
        {
            r = number[i];//记录最大值
        }
    }
    int l = 0;
    while(l<=r)//二分模板
    {
        long long mid = (l+r)/2;
        long long minn = 0;//初始化最小值为0,不要负无穷
        int flag = 0;//标记
        for (int i = 1;i<=n;i++)//区间移动
        {
            sum[i] = sum[i-1] + (number[i]-mid);//计算前缀和
            if(i>=m)//当满足区间长度时
            {
                minn = min(minn,sum[i-m]);//更新之前区间的最小值和(sum[i-m]确保了剩余区间长度>=m)
                if(sum[i]-minn>=0)//如果当前区间已经满足条件
                {
                    flag = 1;//标记
                    break;//退出
                }
            }
        }
        if(flag)
        {
            ans = mid;//记录答案
            l = mid + 1;
        }
        else
        {
            r = mid - 1;
        }
    }
    cout<<ans/100<<endl;//输出别忘了除100
    return 0;
}
