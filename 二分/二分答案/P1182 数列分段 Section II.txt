#include <bits/stdc++.h>
using namespace std;
int n,m;
int number[100001];
int ans;
bool check(int x)
{
    int sum = 0;
    int amount = 0;
    for (int i = 1;i<=n;i++)
    {
        if(sum+number[i]<=x)
        {
            sum += number[i];
        }
        else
        {
            if(number[i]>x) return false;
            sum = number[i];
            amount++;
        }
    }
    amount++;
    return amount<=m;
}
int main()
{
    cin>>n>>m;
    int r = 1e9;
    int l = 0;
    for (int i = 1;i<=n;i++)
    {
        cin>>number[i];
        if(number[i]>l) l = number[i];
    }
    while(l<=r)
    {
        int mid = (l+r)/2;
        if(check(mid))
        {
            ans = mid;
            r = mid - 1;
        }
        else
        {
            l = mid + 1;
        }
    }
    cout<<ans<<endl;
    return 0;
}
