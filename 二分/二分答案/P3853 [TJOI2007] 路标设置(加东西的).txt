#include <bits/stdc++.h>
using namespace std;
int p[100005];
int temp[100005];
int ans;
int length,n,k;
bool check(int x)
{
    int sum = 0;
    memcpy(temp,p,sizeof(p));
    for (int i = 1;i<=n+1;i++)
    {
        while(temp[i]>x)
        {
            temp[i] -= x;
            sum++;
        }
    }
    return sum<=k;
}
int main()
{
    cin>>length>>n>>k;
    int before = 0;
    int a;
    for (int i = 1;i<=n;i++)
    {
        cin>>a;
        p[i] = a-before;
        before = a;
    }
    p[n+1] = length - before;
    int l = 1;//一定要是1
    int r = length;
    while(l<=r)
    {
        int mid = (l+r)/2;
        if(check(mid))
        {
            ans = mid;
            r = mid - 1;
        }
        else
        {
            l = mid + 1;
        }
    }
    cout<<ans<<endl;
    return 0;
}
