#include <bits/stdc++.h>
using namespace std;
int n,m;
int a[1000001];
int b[100001];
int check(int x)
{
    int l = 1;
    int r = n;
    int ans = 0;
    while(l<=r)
    {
        int mid = (l+r)/2;
        if(a[mid]<x)
        {
            l = mid + 1;

        }
        else if(a[mid]>x)
        {
            r = mid - 1;
        }
        else
        {
            ans = mid;
            r = mid - 1;
        }
    }
    if(ans) return ans;
    else return -1;
}
int main()
{
    cin>>n>>m;
    for (int i = 1;i<=n;i++)
    {
        cin>>a[i];
    }
    for (int i = 1;i<=m;i++)
    {
        cin>>b[i];
    }
    int pos;
    for (int i = 1;i<=m;i++)
    {
        pos = check(b[i]);
        cout<<pos<<" ";
    }
    cout<<endl;
    return 0;
}
