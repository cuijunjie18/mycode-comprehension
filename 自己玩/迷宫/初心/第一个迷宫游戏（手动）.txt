#include <stdio.h>
#include <stdlib.h>
#include <Windows.h>
#include <conio.h>
#define N 50
#define M 50
void Show(char str[][M],int n,int m);
void Updatawithinput(char str[][M],int n,int m);
int main()
{
    char str[N][M] =
    {

        "************ ",
        "*o         * ",
        "* ********** ",
        "* *   *   ** ",
        "* * * * * ** ",
        "* * * * *  * ",
        "* * * * *  * ",
        "* * * * ** * ",
        "* * * * ** * ",
        "*   * * ** * ",
        "*   *   *  0 ",
        "************ "
    };
    int n = 12;
    int m = 12;
    Show(str,n,m);
    Updatawithinput(str,n,m);
    return 0;
}

void Show(char str[][M],int n,int m)
{
    for (int i = 0;i<=n-1;i++)
    {
        for (int j = 0;j<=m-1;j++)
        {
            printf("%c",str[i][j]);
        }
        printf("\n");
    }
}

void Updatawithinput(char str[][M],int n,int m)
{
    int x = 1,y = 1;
    int exitx = n-2,exity = m-2;
    char input;
    while(x!=exitx||y!=exity)
    {
        if(kbhit())
        {
            input = getch();
            if(input == 'w'&&str[x-1][y]!='*')
            {
                str[x-1][y] = 'o';
                str[x][y] = ' ';
                x--;
            }
            if(input == 's'&&str[x+1][y]!='*')
            {
                str[x+1][y] = 'o';
                str[x][y] = ' ';
                x++;
            }
            if(input == 'a'&&str[x][y-1]!='*')
            {
                str[x][y-1] = 'o';
                str[x][y] = ' ';
                y--;
            }
            if(input == 'd'&&str[x][y+1]!='*')
            {
                str[x][y+1] = 'o';
                str[x][y] = ' ';
                y++;
            }
        }
        system("cls");
        Show(str,n,m);
        Sleep(50);
    }
    printf("You win!\n");
    system("PAUSE");
}
