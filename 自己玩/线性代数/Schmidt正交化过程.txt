#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int n,m;
double vector[100][100];
void Schmidt(double a[100][100],int n,int m);
double Count(double b[100],double a[100]);
int main()
{
    printf("Please input the number of the vector:");
    scanf("%d",&m);
    printf("Please input the dimension of a vector:");
    scanf("%d",&n);
    for (int i = 1;i<=m;i++)
    {
        printf("Please input the %d vector:",i);
        for (int j = 1;j<=n;j++)
        {
            scanf("%lf",&vector[i][j]);
        }
    }
    printf("The matrix consist of these vector is:\n");
    for (int i = 1;i<=n;i++)
    {
        for (int j = 1;j<=m;j++)
        {
            printf("%.2lf ",vector[j][i]);
        }
        printf("\n");
    }
    printf("\n");
    Schmidt(vector,n,m);
    return 0;
}

void Schmidt(double a[100][100],int n,int m)
{
    double test[100][100];
    memcpy(test,a,sizeof(vector));
    for (int i = 1;i<=m;i++)
    {
        for (int j = 1;j<=i-1;j++)
        {
            double factor = Count(test[j],a[i]);
            for (int k = 1;k<=n;k++)
            {
                test[i][k] += factor*test[j][k];
            }
        }
    }
    printf("The Schmidt transform matrix is:\n");
    for (int i = 1;i<=n;i++)
    {
        for (int j = 1;j<=m;j++)
        {
            printf("%.2lf ",test[j][i]);
        }
        printf("\n");
    }
}

double Count(double b[100],double a[100])
{
    double son = 0,mum = 0;
    for (int i = 1;i<=n;i++)
    {
        mum += b[i]*b[i];
        son += a[i]*b[i];
    }
    return (-son/mum);
}
