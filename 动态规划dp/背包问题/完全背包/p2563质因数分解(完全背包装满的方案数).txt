#include <bits/stdc++.h>
using namespace std;
int f[300];
int number[300];
bool Isprime(int a)
{
    if(a <= 1) return 0;
    if(a == 2||a == 3) return 1;
    if(a%6!=1&&a%6!=5) return 0;
    for (int i = 5;i*i<=a;i += 6)
    {
        if(a%i == 0 || a%(i+2) == 0) return 0;
    }
    return 1;
}
int main()
{
    int a = 0;
    for (int i = 1;i<=200;i++)
    {
        if(Isprime(i))
        {
            a++;
            number[a] = i;
        }
    }
    for (int i = 1;i<=a;i++)
    {
        for (int j = 1;j<=200;j++)
        {
            if(j - number[i] == 0) f[j] += 1;
            else if(j - number[i]>0) f[j] += f[j-number[i]];
        }
    }
    while(scanf("%d",&a)!=EOF)
    {
        printf("%d\n",f[a]);
    }
    return 0;
}
