#include <stdio.h>
#include <stdlib.h>
int f[10001];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    int a[n+1];
    for (int i  =1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    for (int i = 1;i<=n;i++)
    {
        for (int j = m;j>=a[i];j--)
        {
            if(j == a[i])
                f[j] = f[j]+1;
            else
                f[j] = f[j-a[i]]+f[j];
        }
    }
    printf("%d",f[m]);
    return 0;
}
