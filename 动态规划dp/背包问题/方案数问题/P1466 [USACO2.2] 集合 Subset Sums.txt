#include <bits/stdc++.h>
using namespace std;
long long f[100000];//不开long long 没了
int number[50];
int main()
{
    int n,v;
    scanf("%d",&n);
    for (int i = 1;i<=n;i++)
    {
        number[i] = i;
    }
    int sum = (n+1)*n/2;
    if(sum%2!=0)
    {
        printf("0\n");
        return 0;
    }
    else
    {
        v = sum/2;
        for (int i = 1;i<=n;i++)
        {
            for (int j = v;j>=number[i];j--)
            {
                if(j - number[i] == 0) f[j] = f[number[i]] + 1;
                else if(j - number[i]>0) f[j] = f[j-number[i]] + f[j];
            }
        }
    }
    printf("%lld\n",f[v]/2);
    return 0;
}
