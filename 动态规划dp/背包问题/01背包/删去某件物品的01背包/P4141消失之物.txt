//本题是要求装满的方案数矩阵模10
#include <bits/stdc++.h>
using namespace std;
long long f[2001];//long long 防爆
long long g[2001];
int w[2001];//记录体积
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i = 1;i<=n;i++)
    {
        scanf("%d",&w[i]);
    }
    f[0] = 1;//初始化
    for (int i = 1;i<=n;i++)
    {
       for (int j = m;j>=w[i];j--)
       {
           f[j] = (f[j-w[i]] + f[j])%10;//一定要模10，不然可能long long 都会爆
       }
    }
    for (int i = 1;i<=n;i++)
    {
        memcpy(g,f,sizeof(f));
        for (int j = w[i];j<=m;j++)//这里要顺序减去，因为要保证减去的时没有拿过该物品的，自己想想(可以认为每个物品都是dp时的最后一个物品)
        {
            g[j] = (g[j] - g[j-w[i]] + 10)%10;//要加10再模，因为可能模之后相减会负数，所以给前面加个10，因为前面一定大
        }
        for (int j = 1;j<=m;j++) printf("%lld",g[j]%10);
        printf("\n");
    }
    return 0;
}
