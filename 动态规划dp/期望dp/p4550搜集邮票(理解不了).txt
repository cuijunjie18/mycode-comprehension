#include <bits/stdc++.h>
using namespace std;
double f[10001];
double g[10001];
int main()
{
    int n;
    scanf("%d",&n);
    f[n] = 0.0;
    for (int i = n-1;i>=0;i--)
    {
        f[i] = f[i+1] + 1.0*n/(n-i);
    }
    g[n] = 0;
    for (int i = n-1;i>=0;i--)
    {
        g[i] = i*1.0/(n-i)*(f[i] + 1) + g[i+1] + f[i+1] + 1;
    }
    printf("%.2f\n",g[0]);
    return 0;
}
