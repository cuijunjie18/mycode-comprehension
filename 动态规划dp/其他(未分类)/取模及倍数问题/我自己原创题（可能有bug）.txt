给n个数 ，任选（每个数最多选一次），输入d，使得选出的数和为d的倍数，求和的最大值





#include <stdio.h>
#include <stdlib.h>
long long f[101][100];
int number[101];
int yue[101];
long long Max(long long a,long long b)
{
    return a>b?a:b;
}
int main()
{
    int n,k,d;
    scanf("%d%d%d",&n,&k,&d);
    for (int i = 1;i<=n;i++)
    {
        scanf("%d",&number[i]);
        yue[i] = number[i]%d;
    }
    for (int i = 1;i<=n;i++)
    {
        f[i][yue[i]] = number[i];
    }
    for (int i = 2;i<=n;i++)
    {
        for (int j = 0;j<=d-1;j++)
        {
            f[i][j] = Max(f[i][j],f[i-1][(j-yue[i]+d)%d]+number[i]);
        }
    }
    if(f[n][0] == 0)
    {
        printf("-1");
    }
    else
    {
        printf("%lld",f[n][0]);
    }
    return 0;
}
