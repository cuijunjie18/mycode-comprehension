#include <bits/stdc++.h>
using namespace std;
int a[101][101];
int f[20001];
int main()
{
    int s,n,m;
    cin>>s>>n>>m;
    for (int i = 1;i<=s;i++)
    {
        for (int j = 1;j<=n;j++)
        {
            cin>>a[j][i];
        }
    }
    for (int i = 1;i<=n;i++)
    {
        sort(a[i]+1,a[i]+s+1);
    }
    for (int i = 1;i<=n;i++)
    {
        for (int j = m;j>=0;j--)
        {
            for (int k = 1;k<=s;k++)
            {
                if(j>2*a[i][k])
                {
                    f[j] = max(f[j],f[j-2*a[i][k]-1]+k*i);
                }
            }
        }
    }
    cout<<f[m]<<endl;
    return 0;
}
