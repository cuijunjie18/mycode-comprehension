//背景:虽然时间复杂度是较差,但是其交换次数稳定为n-1,故在某些求交换方式的问题中可利用
#include <stdio.h>
#include <stdlib.h>
void Sortdata(int a[],int n)
{
    int pos;
    int temp;
    for (int i = 0;i<=n-2;i++)
    {
        pos = i;
        for (int j = i+1;j<=n-1;j++)
        {
            if(a[j]<a[pos])
            {
                pos = j;
            }
        }
        if(pos!=i)
        {
            temp = a[i];
            a[i] = a[pos];
            a[pos] = temp;
        }
    }
}

void Print(int a[],int n)//打印函数
{
    for (int i = 0;i<=n-1;i++)
    {
        printf("%d\n",a[i]);
    }
}

int main()
{
    int n;
    printf("Please input how many number you want to save:");
    scanf("%d",&n);
    int number[n];
    for (int i = 0;i<=n-1;i++)
    {
        scanf("%d",&number[i]);
    }
    Sortdata(number,n);//排序
    Print(number,n);//打印
    return 0;
}
