//注意事项:只能快读整数，且可以是负数
//原理:getchar 读入比scanf与cin等都要快，但在数据量不大的时候不明显
inline int read()//返回类型根据数据范围定，
{
	int s = 0, w = 1;//s为初始值，w为确定数字的正负
	char ch = getchar();//读入
	while (ch < '0' || ch>'9')//如果不是数字,同时可以去掉之前的空白字符
	{
		if (ch == '-')//如果是符号'-'
			w = -1;//改成-1
		ch = getchar();//继续读入
	}
	while (ch >= '0' && ch <= '9')//读入数字，直到空格或回车退出，代表该数读完了
	{
		s = s * 10 + ch - '0';
		ch = getchar();
	}
	return s*w;//返回数字乘于符合位
}
