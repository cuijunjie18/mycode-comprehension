//背景：暑假补题，cf，第一道交互题
//思路：二分 + 并查集维护
#include <bits/stdc++.h>
#define MAXN 1005
using namespace std;
int f[MAXN];
vector<pair<int,int>> ans;
int Find(int x)
{
    if (x == f[x]) return x;
    else return f[x] = Find(f[x]);
}
void DFS(int a,int b)
{
    if (Find(a) == Find(b)) return;
    cout << "?" << " " << a << " " << b << "\n"; // 询问
    cout.flush();

    int x;
    cin >> x; // 回答，可以知道这个一点是a-b的中点
    if (x == a || x == b) // 中点
    {
        int f1 = Find(a);
        int f2 = Find(b);
        f[f1] = f2;
        ans.push_back({a,b});
        return;
    }
    DFS(a,x); // 二分
    DFS(x,b);
}
void Solve()
{
    ans.clear();
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) f[i] = i; // 并查集初始化
    for (int i = 2; i <= n; i++) DFS(1,i);
    cout << "!" << " ";
    cout.flush();
    for (auto v : ans)
    {
        cout << v.first << " " << v.second << " ";
        cout.flush();
    }
    cout << "\n";
    cout.flush();
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int t = 1;
    cin >> t;
    while (t--)
    {
        Solve();
    }
    return 0;
}
