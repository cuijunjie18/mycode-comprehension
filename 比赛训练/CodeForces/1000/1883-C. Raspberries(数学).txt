//背景：暑假练习，思维题
//思路：k的范围小，2<=k<=5，2,3,5为质数，可以仅加一个数得到，4较特殊，单独讨论即可
#include <bits/stdc++.h>
using namespace std;
void Solve()
{
    int n,k;
    cin>>n>>k;
    int a[n+1];
    int times[5];//记录余数出现次数
    memset(times,0,sizeof(times));
    for (int i = 1;i<=n;i++)
    {
        cin>>a[i];
        times[a[i]%k]++;//余数累加
    }
    if(times[0])//如果有0
    {
        cout<<0<<"\n";//不用操作
        return;
    }
    if(k != 4)//不是4
    {
        for (int i = k-1;i>=1;i--)//倒着找最小操作
        {
            if(times[i])//如果有
            {
                cout<<k-i<<"\n";//输出
                return;
            }
        }
    }
    else//如果是4，细节多，情况多
    {
        if(times[2] >= 2)//余2的超过2个
        {
            cout<<0<<"\n";//不用操作
        }
        else if(times[3]||(times[2] == 1&&times[1]))//操作为1
        {
            cout<<1<<"\n";
        }
        else if(times[2]||times[1]>=2)//操作为2
        {
            cout<<2<<"\n";
        }
        else cout<<3<<"\n";//其他操作为3
    }
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int t = 1;
    cin>>t;
    while(t--)
    {
        Solve();
    }
    return 0;
}
