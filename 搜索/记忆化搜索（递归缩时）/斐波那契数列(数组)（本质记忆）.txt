#include <stdio.h>
#define N 40
void Fbnq(int n,int FBNQ[])
{
    int i;
    for(i = 0;i<=n-1;i++)
    {
        if(i==0)
        {
            FBNQ[i] = 1;
        }
        else if(i==1)
        {
            FBNQ[i] = 1;
        }
        else
        {
            FBNQ[i] = FBNQ[i-1] + FBNQ[i-2];
        }
    }
}

int main()
{
    int FBNQ[N];
    int n;
    printf("Please input the number of the sequence:");
    scanf("%d",&n);
    Fbnq(n,FBNQ);
    printf("%d",FBNQ[n-1]);
    return 0;
}
