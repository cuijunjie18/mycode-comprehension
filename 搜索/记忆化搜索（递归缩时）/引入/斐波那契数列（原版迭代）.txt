#include <stdio.h>
#include <stdlib.h>
int main()
{
    int n;
    printf("Please input the n:");
    scanf("%d",&n);
    long long sum1 = 1,sum2 = 1,sum3 = 0;
    for (int i = 3;i<=n;i++)
    {
        sum3 = sum1 + sum2;
        sum1 = sum2;
        sum2 = sum3;
    }
    if(n==1)
        printf("1");
    else if(n==2)
        printf("1");
    else
        printf("%lld",sum3);
}
