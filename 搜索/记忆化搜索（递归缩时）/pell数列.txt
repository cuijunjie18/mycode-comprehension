#include <stdio.h>
#include <stdlib.h>
#define N 10000
long long save[N];
long long ABC(int n)
{
    if(save[n]!=0) return save[n];
    else
    {
        if(n==1)
        {
            save[1] = 1;
            return 1;
        }
        else if(n==2)
        {
            save[2] = 2;
            return 2;
        }
        else
        {
            return save[n] = 2*ABC(n-1) + ABC(n-2);
        }
    }1
}
int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for (int i = 0;i<=n-1;i++)
    {
        scanf("%d",&a[i]);
    }
    for (int i = 0;i<=n-1;i++)
    {
        printf("%lld\n",ABC(a[i]));
    }
    return 0;
}
