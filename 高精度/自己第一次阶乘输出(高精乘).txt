#include <stdio.h>
#include <stdlib.h>
#define N 1000
int number[N];
int main()
{
    int wei = 1;
    int k;
    number[1] = 1;
    int n;
    printf("Please input n:");
    scanf("%d",&n);
    for (int i = 1;i<=n;i++)
    {
        for (int j = 1;j<=wei;j++)
        {
            number[j] = number[j] * i;
        }
        for (k = 1;k<wei;k++)
        {
            if(number[k]>=10)
            {
                number[k+1] = number[k]/10 + number[k+1];
                number[k] = number[k]%10;
            }
        }
        while(number[wei]>=10)
        {
            number[wei+1] = number[wei]/10 + number[wei+1];
            number[wei] = number[wei]%10;
            wei++;
        }
        printf("%d! = ",i);
        for (int i = wei;i>=1;i--)
        {
            printf("%d",number[i]);
        }
        printf("\n");
    }
    return 0;
}
